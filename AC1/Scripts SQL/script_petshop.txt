CREATE TABLE pet (
  id varchar (6) PRIMARY KEY,
  nome varchar (50),
  especie varchar (10),
  data_nascimento date
);
  
CREATE TABLE cliente (
  cpf varchar (11) PRIMARY KEY,
  nome varchar (50),
  telefone varchar (11),
  email varchar (100),
  id_pet varchar (6),
  FOREIGN KEY(id_pet) REFERENCES pet(id)
);
